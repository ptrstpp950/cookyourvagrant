Add chef to vagrantfile

In vagrant file include below:


  config.vm.box = "windows2012"
  config.vm.network "private_network", ip: "192.168.33.16"
  config.vm.provision "chef_solo" do |chef|
    chef.cookbooks_path = "cookbooks"
    chef.add_recipe "iis"
    chef.add_recipe "git::windows"
    chef.add_recipe "mywebsite"
  end

And we are ready to run

vagrant up
