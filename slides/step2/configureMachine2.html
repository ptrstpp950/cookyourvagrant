  config.vm.define "nimbus" do |nimbus|
    nimbus.vm.network "private_network", ip: "192.168.33.10"
    nimbus.vm.hostname = "nimbus"
    nimbus.vm.provision "chef_solo" do |chef|
      chef.cookbooks_path = "cookbooks"
    end
  end
