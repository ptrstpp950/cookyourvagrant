Setup storm supervisor

In file stormSupervisor.rb code is 100% analogues to nimbus recipe, only filename changed. The template file you can download from here

include_recipe "storm::stormCommon"

template "/etc/supervisord.d/supervisord-storm-supervisor.ini" do
	source "supervisord-storm-supervisor.ini"
	owner "root"
	group "root"
	mode "0644"
end

service "supervisord" do
	supports :status => true, :restart => true, :reload => true
	action [ :enable, :restart ]
end
